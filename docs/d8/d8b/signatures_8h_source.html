70 /*!
71  * \brief Calculates the \p digest for the specified \p context, clearing the \p context afterwards.
72  *
73  * Transforms the remaining input, if any, and finishes the message with proper padding. Calculates the digest, and clears the context.
74  *
75  * \param[inout] context The context to use.
76  * \param[in] data Remaining input to transform. May be `NULL` if there is nothing to transform (however, \p size must be set to `0`).
77  * \param[in] size The size of the remaining input.
78  * \param[out] digest The output location of the digest.
79  *
80  * \return
81  * `#LS_E_NULL` if \p context or \p digest is `NULL`. \n
82  * `#LS_E_NULL` if \p size is greater than `0` and \p input is `NULL`. \n
83  * `#LS_E_SIZE` if \p size is greater than the algorithm's block size. \n
84  * `#LS_E_FAILURE` if any transformations fail. \n
85  * `#LS_E_SUCCESS` otherwise.
86  */
87 typedef ls_result_t (*lssig_hash_finish)(void *const restrict context, const uint8_t *const restrict data, const size_t size, uint8_t *const restrict digest);
